Dictionary<NodeID, NodeID> parentMap;
void BuildParentMap()
{
    parentMap = new();

    foreach (var r in relations)
        parentMap[r.child] = r.parent;
}

void ReconstructNodes(Dictionary<NodeID, AINode> nodes)
{
    foreach (var key in nodes.Keys.ToList())
    {
        var node = nodes[key];

        if (!node.enabled) continue;

        NodeID current = node.id;

        while (parentMap.ContainsKey(current))
        {
            NodeID parent = parentMap[current];

            if (!nodes[parent].enabled)
            {
                node.enabled = false;
                nodes[key] = node; // structなので再代入必須
                break;
            }

            current = parent;
        }
    }
}

void Awake()
{
    BuildParentMap();

    var nodeDict = nodes.ToDictionary(n => n.id, n => n);

    ReconstructNodes(nodeDict);

    foreach (var node in nodeDict.Values)
    {
        if (!node.enabled) continue;

        if (Mathf.Abs(node.weight) < 0.01f) continue;

        ModComponent(node.id);
    }
